app.py

import streamlit as st
import pandas as pd
import numpy as np
import joblib
import pickle
import shap
import matplotlib.pyplot as plt

# Load model and scaler
gb = joblib.load('gradient_boosting_model.pkl')
scaler = joblib.load('scaler.pkl')

with open('train_test_data.pkl', 'rb') as f:
    X_train, X_test, y_train, y_test = pickle.load(f)

st.set_page_config(page_title="Bank Churn Risk", layout="wide")
st.title("Bank Customer Churn Risk System")

# Sidebar inputs
st.sidebar.header("Customer Profile")
credit_score = st.sidebar.slider("Credit Score", 300, 850, 650)
age = st.sidebar.slider("Age", 18, 92, 40)
tenure = st.sidebar.slider("Tenure (Years)", 0, 10, 5)
balance = st.sidebar.number_input("Balance", 0.0, 250000.0, 50000.0)
num_products = st.sidebar.selectbox("Number of Products", [1, 2, 3, 4])
has_cr_card = st.sidebar.selectbox("Has Credit Card", [0, 1])
is_active = st.sidebar.selectbox("Is Active Member", [0, 1])
salary = st.sidebar.number_input("Estimated Salary", 0.0, 200000.0, 100000.0)
geography = st.sidebar.selectbox("Geography", ["France", "Germany", "Spain"])
gender = st.sidebar.selectbox("Gender", ["Male", "Female"])

# Feature engineering
balance_salary_ratio = balance / (salary + 1)
product_density = num_products / (tenure + 1)
engagement_product = is_active * num_products
age_tenure = age * tenure

geo_germany = 1 if geography == "Germany" else 0
geo_spain = 1 if geography == "Spain" else 0
gender_male = 1 if gender == "Male" else 0

scale_cols = ['CreditScore', 'Age', 'Tenure', 'Balance',
              'EstimatedSalary', 'Balance_Salary_Ratio',
              'Product_Density', 'Age_Tenure']

input_df = pd.DataFrame([{
    'CreditScore': credit_score,
    'Age': age,
    'Tenure': tenure,
    'Balance': balance,
    'NumOfProducts': num_products,
    'HasCrCard': has_cr_card,
    'IsActiveMember': is_active,
    'EstimatedSalary': salary,
    'Product_Density': product_density,
    'Engagement_Product': engagement_product,
    'Balance_Salary_Ratio': balance_salary_ratio,
    'Age_Tenure': age_tenure,
    'Geography_Germany': geo_germany,
    'Geography_Spain': geo_spain,
    'Gender_Male': gender_male
}])
input_df[scale_cols] = scaler.transform(input_df[scale_cols])

# Module 1 - Churn Risk Calculator
st.header("Churn Risk Score")
prob = gb.predict_proba(input_df)[0][1]
flag = "HIGH RISK" if prob >= 0.5 else "LOW RISK"
color = "red" if prob >= 0.5 else "green"

st.metric("Churn Probability", f"{prob:.2%}")
st.markdown(f"### Status: :{color}[{flag}]")

# Module 2 - Probability Distribution
st.header("Churn Probability Distribution")
all_probs = gb.predict_proba(X_test)[:, 1]
fig, ax = plt.subplots()
ax.hist(all_probs, bins=40, color='steelblue', edgecolor='black', alpha=0.7)
ax.axvline(prob, color='red', linestyle='--', label=f'This Customer: {prob:.2%}')
ax.set_xlabel("Churn Probability")
ax.set_ylabel("Number of Customers")
ax.legend()
st.pyplot(fig)

# Module 3 - Feature Importance
st.header("Feature Importance")
importances = gb.feature_importances_
feat_df = pd.DataFrame({
    'Feature': input_df.columns,
    'Importance': importances
}).sort_values('Importance', ascending=False)

fig2, ax2 = plt.subplots(figsize=(10, 5))
ax2.barh(feat_df['Feature'], feat_df['Importance'], color='coral', edgecolor='black')
ax2.invert_yaxis()
ax2.set_title("Feature Importance — Gradient Boosting")
st.pyplot(fig2)

# Module 4 - What-If Simulator
st.header("What-If Scenario Simulator")
st.write("Adjust the sidebar values and observe how churn probability changes in real time.")
st.info(f"Current churn probability: **{prob:.2%}** — tweak Age, Products, or Active Member status to see impact.")
